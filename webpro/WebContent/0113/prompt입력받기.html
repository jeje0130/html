script에서 HTML요소에 접근
	방법1) document.getElementsByTagName("div")[0];
	      document.getElementsById('aa') : #, . 표기 안해줘도됨
	
	//ECMAScript(es6)버전
	방법2) document.querySelector('div')   :  요소가 여러개가 있어도 첫번째 요소로 접근
          document.querySelector('#aa')   : #아이디 표기해줘야함
          document.querySelector('.aa')   : .클래스 표기해줘야함
	
	방법3) document.querySelectorAll('div')   :  모든 div 접근
